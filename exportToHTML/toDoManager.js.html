17   ToDo.prototype.createRepetition = function () { 
18       var copy = new ToDo(this.description, this.dueDate, this.repetition); 
19       copy.copiedFromId = this.id; 
20    
21       if (copy.repetition == "") { 
22           return copy; 
23       } 
24    
25       var date = new Date(); 
26       if (this.repeatFrom == "d") { 
27           date.fromStringSortable(this.dueDate) 
28       } 
29    
30       switch (this.repeatIntervalType) { 
31           case 'd': 
32               date.addDays(this.repeatIntervalNumber); 
33               break; 
34           case 'w': 
35               date.addWeeks(this.repeatIntervalNumber); 
36               break; 
37           case 'm': 
38               date.addMonth(this.repeatIntervalNumber); 
39               break; 
40           case 'y': 
41               date.addYears(this.repeatIntervalNumber); 
42               break; 
43       } 
44    
45       copy.dueDate = date.toStringSortable(); 
46       copy.dueDateDisplay = copy.getDueDateDisplay(copy.dueDate); 
47    
48       return copy; 
49   };

159  function getDueSectionName(dueDate) { 
160      if (dueDate == "") { 
161          return 'noDueDate'; 
162      } 
163      else if (dueDate < today) { 
164          return 'overdue'; 
165      } 
166      else if (dueDate == today) { 
167          return 'today'; 
168      } 
169      else if (dueDate == tomorrow) { 
170          return 'tomorrow'; 
171      } 
172      else if (dueDate == dayAfterTomorrow) { 
173          return 'dayAfterTomorrow'; 
174      } 
175      else { 
176          return 'future'; 
177      } 
178  }

196  function sortFunction(a, b) { 
197   
198      if (a['dueDate'] == b['dueDate']) { 
199          if (a['description'] == b['description']) { 
200              return 0; 
201          } 
202          return a['description'] < b['description'] ? -1 : +1; 
203      } 
204   
205      if (a['dueDate'] == "") { 
206          return +1; 
207      } 
208   
209      if (b['dueDate'] == "") { 
210          return -1; 
211      } 
212   
213      return a['dueDate'] < b['dueDate'] ? -1 : +1; 
214  }

380      var descriptionInput = window.document.getElementById("descriptionInput"); 
381      var dueDateInput = window.document.getElementById("dueDateInput"); 
382      var intervalNumberSelection = window.document.getElementById("intervalNumberSelection"); 
383      var intervalTypeSelection = window.document.getElementById("intervalTypeSelection"); 
384      var repeatFromSelection = window.document.getElementById("repeatFromSelection"); 
385   
386      var intervalTypes = [" d", " w", " m", " y"]; 
387      var repeatFromTypes = [" c", " d"] 
388   
389      if (descriptionInput.value == "") { 
390          return; 
391      } 
392   
393      var repetition = ""; 
394      if (dueDateInput.value != "" && intervalNumberSelection.selectedIndex != 0) { 
395          repetition += intervalNumberSelection.selectedIndex; 
396          repetition += intervalTypes[intervalTypeSelection.selectedIndex]; 
397          repetition += repeatFromTypes[repeatFromSelection.selectedIndex]; 
398      } 
399   
400      var convertedDate = ""; 
401      if (dueDateInput.value != "") { 
402          convertedDate = convertDisplayDateToSortableDate(dueDateInput.value); 
403      } 
404   
405      var toDo = new ToDo(descriptionInput.value, convertedDate, repetition); 
406   
407      toDos.push(toDo); 
408      generateToDo(toDo); 
409      checkSections(); 
410   
411      descriptionInput.value = ""; 
412      dueDateInput.value = ""; 
413      intervalNumberSelection.selectedIndex = 0; 
414      intervalTypeSelection.selectedIndex = 0; 
415      repeatFromSelection.selectedIndex = 0; 
416      descriptionChanged();
