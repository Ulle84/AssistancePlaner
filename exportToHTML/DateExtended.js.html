3    Date.prototype.addDays = function (number) { 
4        var year = this.getFullYear(); 
5        var month = this.getMonthCorrected(); 
6        var day = this.getDate(); 
7     
8        for (var i = 0; i < number; i++) { 
9            day++; 
10           if (day > this.getNumberOfDays(year, month)) { 
11               day = 1; 
12               month++; 
13               if (month == 13) { 
14                   month = 1; 
15                   year++; 
16               } 
17           } 
18       } 
19    
20       this.setFullYear(year); 
21       this.setMonth(month - 1); 
22       this.setDate(day); 
23   }

102      var day = this.getDate(); 
103   
104      var result = ""; 
105      if (day < 10) { 
106          result += "0";

109      result += "."; 
110   
111      if (month < 10) { 
112          result += "0";

126  Date.prototype.isInLeapYear = function () { 
127      var year = this.getFullYear(); 
128      return ((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0); 
129  }

131  Date.prototype.getNumberOfDaysInThisMonth = function () { 
132      var month = this.getMonthCorrected(); 
133      return /4|6|9|11/.test(month) ? 30 : month == 2 ? this.isInLeapYear() ? 29 : 28 : 31; 
134  }

136  Date.prototype.getNumberOfDays = function (year, month) { 
137      return /4|6|9|11/.test(month) ? 30 : month == 2 ? ((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0) ? 29 : 28 : 31; 
138  }
