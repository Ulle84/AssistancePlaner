24   function createContents() { 
25       contents = window.document.createElement("div"); 
26       documentationSection.insertBefore(contents, documentationSection.firstChild); 
27        
28       if (window.document.getElementsByTagName("h1").length == 0) { 
29           return; 
30       } 
31        
32       contents.setAttribute("class", "contents"); 
33        
34       var a = window.document.createElement("a"); 
35       a.setAttribute("name", navigationLinkBaseName + counter); 
36       contents.appendChild(a); 
37       counter++; 
38    
39       var contentsHeading = window.document.createElement("div"); 
40       contentsHeading.textContent = "Inhaltsverzeichnis"; 
41       contentsHeading.setAttribute("class", "contentsHeading"); 
42       contents.appendChild(contentsHeading); 
43    
44       var headerNumber = null; 
45       for (var i = 0; i < documentationSectionChilds.length; i++) { 
46           switch(documentationSectionChilds[i].nodeName.toUpperCase()) { 
47           case "H1": 
48           case "H2": 
49           case "H3": 
50           case "H4": 
51           case "H5": 
52           case "H6": 
53               createNewContentsEntry(documentationSectionChilds[i].nodeName, documentationSectionChilds[i]); 
54               break; 
55           } 
56       } 
57   }

59   function createTitle() { 
60       var titles = window.document.getElementsByTagName("title"); 
61        
62       if (titles.length == 0) { 
63           return; 
64       } 
65        
66       var title = titles[0]; 
67        
68       var div = window.document.createElement("div"); 
69       documentationSection.insertBefore(div, documentationSection.firstChild); 
70        
71       div.setAttribute("class", "title"); 
72       div.textContent = title.textContent; 
73   }

75   function createNewContentsEntry(tagname, node) { 
76       // create headerNumber 
77       var level = parseInt(tagname[1]); 
78       headerNumber = ""; 
79       headerCounters[level - 1]++; 
80       for (var i = level; i < headerCounters.length; i++) { 
81           headerCounters[i] = 0; 
82       } 
83       for (var i = 0; i < level; i++) { 
84           headerNumber += headerCounters[i] 
85           if (i != level -1) { 
86               headerNumber += headerSeperator; 
87           } 
88       } 
89    
90       // create navigation element 
91       var newDiv = window.document.createElement("div"); 
92       var a = window.document.createElement("a"); 
93       newDiv.appendChild(a); 
94       newDiv.setAttribute("class", "contentsH" + level); 
95       a.textContent = headerNumber + " " +  node.textContent; 
96       contents.appendChild(newDiv); 
97        
98       // set headerNumber 
99       node.textContent = headerNumber + " " + node.textContent; 
100       
101      // set link for internal navigation 
102      var navigationLink = navigationLinkBaseName + counter; 
103      a.setAttribute("href", "#" + navigationLink) 
104      var a2 = window.document.createElement("a"); 
105      a2.setAttribute("name", navigationLink); 
106      node.insertBefore(a2, node.firstChild); 
107       
108      // find references and link them 
109      var id = node.getAttribute("id"); 
110      if (id != null) { 
111          var links = window.document.getElementsByClassName(id); 
112          for (var i = 0; i < links.length; i++) { 
113              var link = links[i]; 
114              var nobr = window.document.createElement("nobr"); 
115              var a3 = window.document.createElement("a"); 
116              a3.setAttribute("href", "#" + navigationLink); 
117              a3.textContent = "Kapitel " + headerNumber; 
118              link.textContent = ""; 
119              nobr.appendChild(a3); 
120              link.appendChild(nobr); 
121          } 
122      } 
123      counter++; 
124  }

126  function initVariables() { 
127      documentationSection = window.document.getElementById("documentation"); 
128      documentationSectionChilds = documentationSection.childNodes; 
129      headerCounters = new Array(0, 0, 0, 0, 0, 0); 
130  }

132  function createTableToolTips() { 
133      var tables = window.document.getElementsByTagName("table"); 
134      for(var i = 0; i < tables.length; i++) { 
135          var table = tables[i]; 
136          var tr = table.getElementsByTagName("tr"); 
137           
138          var th = tr[0].getElementsByTagName("th"); 
139           
140          if (tr.length < 2) { 
141              return; 
142          } 
143          for (var j = 1; j < tr.length; j++) { 
144              var td = tr[j].getElementsByTagName("td"); 
145              for (var k = 0; k < th.length; k++) { 
146                  td[k].setAttribute("title", th[k].textContent); 
147              } 
148          } 
149           
150      } 
151  }

153  function createImageCaptions() { 
154      var images = window.document.getElementsByTagName("img"); 
155      for(var i = 0; i < images.length; i++) { 
156          var img = images[i]; 
157           
158          var divImage = window.document.createElement("div"); 
159          divImage.setAttribute("class", "image"); 
160           
161          var divImageCaption = window.document.createElement("div"); 
162          divImageCaption.setAttribute("class", "imageCaption"); 
163           
164          var divImageCaptionTextContent = "Abbildung " + imageCounter; 
165           
166          var title = img.getAttribute("title"); 
167          if (title != null) { 
168              if (title.length > 0) { 
169                  divImageCaptionTextContent += ": "; 
170                  divImageCaptionTextContent += title; 
171              } 
172          } 
173           
174          divImageCaption.textContent = divImageCaptionTextContent; 
175           
176          img.parentNode.insertBefore(divImage, img); 
177          divImage.appendChild(img); 
178           
179          divImage.appendChild(divImageCaption); 
180           
181          // links to image 
182          var id = img.getAttribute("id");         
183          if (id != null) { 
184              var a2 = window.document.createElement("a"); 
185              a2.setAttribute("name", id); 
186              divImage.insertBefore(a2, divImage.firstChild); 
187               
188              var spans = window.document.getElementsByClassName(id); 
189              for (var j = 0; j < spans.length; j++) { 
190                  spans[j].textContent = ""; 
191                  var a = window.document.createElement("a"); 
192                  var nobr = window.document.createElement("nobr"); 
193                  a.setAttribute("href", "#" + id); 
194                  a.textContent = "Abbildung " + imageCounter; 
195                  nobr.appendChild(a); 
196                  spans[j].appendChild(nobr); 
197              } 
198          } 
199           
200          imageCounter++; 
201      } 
202  }
