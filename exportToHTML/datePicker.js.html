28   function arrayToTable(array, width, firstRowIsHeader) { 
29       var table = window.document.createElement("table"); 
30    
31       var row; 
32       for (var i = 0; i < array.length; i++) { 
33           if (i % width == 0) { 
34               row = window.document.createElement("tr"); 
35               table.appendChild(row); 
36           } 
37           var cell = window.document.createElement((firstRowIsHeader && i < width) ? "th" : "td"); 
38           cell.textContent = array[i]; 
39           if (parseInt(array[i]) > 0) { 
40               cell.setAttribute("onclick", "dateClicked(" + dp_displayYear + ", " + dp_displayMonth + ", " + array[i] + ")"); 
41           } 
42           row.appendChild(cell); 
43       } 
44       return table; 
45   }

57   function createMonthArray(year, month) { 
58       var date = new Date(year, month - 1, 1); 
59       var numberOfDays = date.getNumberOfDaysInThisMonth(); 
60    
61       var firstDay = date.getDay(); // So = 0; Mo = 1; Di = 2 ... 
62       var firstDayOffset = (firstDay + 6) % 7; 
63    
64       var totalCount = firstDayOffset + numberOfDays; 
65    
66       var rowCount = totalCount / 7; 
67       if (totalCount % 7 != 0) { 
68           rowCount++; 
69       } 
70    
71       var monthArray = ["Mo", "Di", "Mi", "Do", "Fr", "Sa", "So"]; 
72    
73       for (var i = 0; i < firstDayOffset; i++) { 
74           monthArray.push(""); 
75       } 
76    
77       for (var i = 1; i <= numberOfDays; i++) { 
78           monthArray.push(i); 
79       } 
80    
81       while (monthArray.length % 7 != 0) { 
82           monthArray.push(""); 
83       } 
84    
85       return monthArray; 
86   }
