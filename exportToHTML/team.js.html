127  function checkLoginNames() { 
128      var teamTable = window.document.getElementById("team") 
129      var rows = teamTable.getElementsByTagName("tr"); 
130   
131      var loginNames = []; 
132      for (var i = 1; i < rows.length; i++) { 
133          var data = rows[i].getElementsByTagName("td"); 
134          if (data[0].firstChild.value == "") { 
135              alert("Eine leere Kennung ist nicht erlaubt!"); 
136              return false; 
137          } 
138          loginNames.push(data[0].firstChild.value); 
139      } 
140   
141      loginNames.sort(); 
142      var last = loginNames[0]; 
143      for (var i = 1; i < loginNames.length; i++) { 
144          if (loginNames[i] == last) { 
145              alert("Die Tabelle kann nicht gespeichert werden!\nDie Kennung " + last + " ist doppelt enthalten."); 
146              return false; 
147          } 
148          last = loginNames[i]; 
149      } 
150      return true; 
151   
152  }

169   
170      for (var i = 1; i < rows.length; i++) { 
171          var data = rows[i].getElementsByTagName("td"); 
172          for (var j = 0; j < data.length; j++) { 
173              if (j < 8) { 
174                  content += data[j].firstChild.value + "\n"; 
175              } 
176              if (j == 8) { 
177                  var checkBoxes = data[j].getElementsByTagName("input"); 
178                  var contentCheckBoxes = ""; 
179                  for (var k = 0; k < checkBoxes.length; k++) { 
180                      if (contentCheckBoxes != "") { 
181                          contentCheckBoxes += ";"; 
182                      } 
183                      if (checkBoxes[k].checked == true) { 
184                          contentCheckBoxes += 1; 
185                      } 
186                      else { 
187                          contentCheckBoxes += 0; 
188                      } 
189   
190                  } 
191                  content += contentCheckBoxes + "\n"; 
192              } 
193          } 
194      }

198   
199      xmlhttp.onreadystatechange = function () {
