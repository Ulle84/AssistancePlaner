95       private function compareTimes($time1, $time2) 
96       { 
97           // returns 0 if equal 
98           // returns 1 if $time1 is before $time2 
99           // returns -1 if $time1 is after $time2 
100   
101          if ($time1 == $time2) { 
102              return 0; 
103          } 
104   
105          $hours1 = substr($time1, 0, 2); 
106          $minutes1 = substr($time1, 3, 2); 
107   
108          $hours2 = substr($time2, 0, 2); 
109          $minutes2 = substr($time2, 3, 2); 
110   
111          if ($hours1 < $hours2) { 
112              return 1; 
113          } 
114   
115          if ($hours1 > $hours2) { 
116              return -1; 
117          } 
118   
119          // here $hours1 should be equal $hours2 
120          if ($minutes1 < $minutes2) { 
121              return 1; 
122          } else { 
123              return -1; 
124          } 
125      }

136      private function addOneHour($time) 
137      { 
138          $hours = (int)substr($time, 0, 2); 
139   
140          $hours++; 
141          if ($hours == 24) { 
142              $hours = 0; 
143          } 
144   
145          $returnTime = ""; 
146          if ($hours < 10) { 
147              $returnTime = "0"; 
148          } 
149          $returnTime .= $hours . substr($time, 2, 3); 
150   
151          return $returnTime; 
152      }

154      private function getServiceHours($day) 
155      { 
156          //substr($this->serviceBegin, 3, 2) 
157          //$hourOfServiceBeg$this->monthPlan->days[$day]->serviceBegin 
158   
159          $serviceHours = "10:00 - 11:00"; 
160   
161          if ($day != 1) { 
162              if ($this->standbyPerson[$day] == $this->servicePerson[$day - 1]) { 
163                  $latterTime = $this->getLatterTime("10:00", $this->monthPlan->days[$day - 1]->serviceEnd); 
164                  $serviceHours = $latterTime . " - " . $this->addOneHour($latterTime); 
165              } 
166          } else { 
167              $previousMonth = $this->month - 1; 
168              $year = $this->year; 
169              if ($previousMonth == 0) { 
170                  $previousMonth = 12; 
171                  $year--; 
172              } 
173   
174              $rosterPreviousMonth = new Roster($year, $previousMonth, $this->rosterAlgorithmVersion); 
175              if ($rosterPreviousMonth->rosterExist) { 
176                  $lastDayOfMonth = date("t", mktime(0, 0, 0, $previousMonth, 1, $year)); 
177                  if ($this->standbyPerson[$day] == $rosterPreviousMonth->servicePerson[$lastDayOfMonth]) { 
178                      $latterTime = $this->getLatterTime("10:00", $rosterPreviousMonth->monthPlan->days[$lastDayOfMonth]->serviceEnd); 
179                      $serviceHours = $latterTime . " - " . $this->addOneHour($latterTime); 
180                  } 
181              } 
182          } 
183   
184          if ($this->monthPlan->days[$day]->serviceHours > 13) { 
185              $serviceHours .= " und 18:00 - 19:00"; 
186          } 
187   
188          return $serviceHours; 
189      }
